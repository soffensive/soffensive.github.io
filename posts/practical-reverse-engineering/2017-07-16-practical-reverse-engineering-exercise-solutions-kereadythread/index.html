VOID KeReadyThread (__inout PKTHREAD Thread)
{
    if (Thread->ApcState.Process->StackCount != 7) {
        KiFastReadythread(Thread); 
    }
    else {
        if (KiInSwapSingleProcess(Thread) == 0) {
            KiFastReadythread(Thread); 
        }
    } 
}

VOID KeReadyThread (__inout PKTHREAD Thread)
{
    if (Thread->ApcState.Process->StackCount == 7) {

            return;
        }
    }

    KiFastReadythread(Thread);
